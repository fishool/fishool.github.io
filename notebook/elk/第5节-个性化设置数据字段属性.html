                                                                      # index 是否可以检索

                                                                      • 是否创建索引, 字段可以被检索, 默认为true

                                                                      应用场景: 公司的概述信息, 商品描述详情, 一般不需要检索, 只需要存储, 所以设置为false 官方文档open in new window

                                                                      DELETE /foolish-test-0001
PUT /foolish-test-0001
{
  "mappings": {
    "properties": {
      "name": {
        "type": "text"
      },
        "age": {
            "type": "integer"
        },
        "description": {
            "index": false,
            "type": "text"
        }
    }
  }
}

# 插入数据
POST /foolish-test-0001/_doc
{
  "name": "fishool",
  "age": 18,
  "description": "this is a description"
}

# 检索数据
GET /foolish-test-0001/_search
{
  "query": {
    "match": {
      "description": "this is a description"
    }
  }
}

                                                                      # store 是否存储

                                                                      • ES会存储原始数据在_source中, 除此歪 Lucene中也可以存储
                                                                      • store: 属性关键字
                                                                      • 默认为false, 不存储
                                                                      • stored_fields: 查询时指定返回存储的字段 官方文档open in new window
                                                                      DELETE /foolish-test-0001
PUT /foolish-test-0001
{
  "mappings": {
    "_source": {
      "enabled": false
    },
    "properties": {
      "name": {
        "type": "text"
      },
        "age": {
             "store": true,
            "type": "integer"
        },
        "description": {
            "store": true,
            "type": "text"
        }
    }
  }
}

# 插入数据
POST /foolish-test-0001/_doc
{
  "name": "fishool",
  "age": 18,
  "description": "this is a description"
}

# 检索数据
GET /foolish-test-0001/_search
{
  "stored_fields": ["*"]
}

                                                                      # enabled 是否启用

                                                                      • object类型的字段, 指定对象下的字段是否创建索引, 默认启动 官方文档open in new window

        "type": "text"

            "enabled": false,
            "properties": {
                "province": {
                    "type": "text"
                },
                "city": {
                    "type": "text"
                }
            }
        }
    }
  }
}

                                                                      # null_value 空值设置

                                                                      • null_value: 空值索引设置, 主要功能是为了占用索引, 方便聚合,搜索
                                                                      • 需要和数据库的null区分 , es 只有在字段值为null时才会索引null_value

                                                                      官方文档open in new window

                                                                      DELETE /foolish-test-0001
PUT /foolish-test-0001
{
  "mappings": {
    "properties": {
      "name": {
        "type": "text"
      },
        "age": {
            "null_value": 0,
            "type": "integer"
        },
        "description": {
            "type": "text"
        }
    }
  }
}

# 插入数据
POST /foolish-test-0001/_doc
{
  "name": "fishool18",
  "age": 18,
  "description": "this is a description"
}

POST /foolish-test-0001/_doc
{
  "name": "fishool19",
  "description": "this is a description"
}

POST /foolish-test-0001/_doc
{
  "name": "fishool_null",
  "age": null,
  "description": "this is a description"
}

                                                                      用于描述 子字段对象类型

                                                                      # dynamic 子对象动态扩展

                                                                      • dynamic: true, 默认为true, 表示动态扩展子对象
                                                                      • dynamic: false, 新字段仅仅存储到原始文档中, 不会创建索引
                                                                      • dynamic: strict, 新字段会报错

                                                                      官方文档open in new window

                                                                      DELETE /foolish-test-0001
PUT /foolish-test-0001
{
  "mappings": {
    "dynamic": "strict",
    "properties": {
      "name": {
        "type": "text"
      },
        "age": {
            "type": "integer"
        },
        "description": {
            "type": "text"
        }
    }
  }
}

# 插入数据
POST /foolish-test-0001/_doc
{
  "name": "fishool18",
  "age": 18,
  "description": "this is a description",
  "xkey": "xvalue"
}

                                                                      # ignore_above 字符超长忽略

                                                                      • ignore_above: 256, 默认为256, 表示超过256个字符的字段不会被索引 等同于index: false 官方文档open in new window

                                                                      # coerce 数值格式强制约束

                                                                      官方文档open in new window

                                                                      coerce 强制进行类型转换, 默认为true , 可以将字符串转存入到数值类型的字段中 设置为false, 存储时会报错

                                                                      DELETE /foolish-test-0001
PUT /foolish-test-0001
{
  "mappings": {
    "properties": {
      "name": {
        "type": "text"
      },
        "age": { 
            "coerce": false,
            "type": "integer"
        },
        "weight": { 
            "type": "integer"
        },
        "description": {
            "type": "text"
        }
    }
  }
}

                                                                      # 高级字段属性

                                                                      1. meta 元数据 官方文档open in new window
                                                                      • 用于添加一些额外的信息, 如作者 , 索引描述等

                                                                      1. analyzer 分词器 官方文档open in new window

                                                                      2. similarity 相似度算法 官方文档open in new window

                                                                      3. boost 权重 官方文档open in new window

                                                                      4. norms 归一化 官方文档open in new window

                                                                      5. normalizer 归一化器 官方文档open in new window

                                                                      6. eager_global_ordinals 提前全局排序 官方文档open in new window

                                                                      • 全局序号 , 用于字符串类聚合统计 , keyword类型采用. 用于提前排序, 提高聚合统计性能.
                                                                      • 默认不启用, 可以通过设置eager_global_ordinals: true 来启用
                                                                      • 当字段类型为keyword时, 会自动启用 , 生成大量的全局序号.
